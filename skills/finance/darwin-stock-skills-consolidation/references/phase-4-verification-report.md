def is_trading_day(date_str):
    """判断是否为交易日"""
    from datetime import datetime
    date = datetime.strptime(date_str, '%Y-%m-%d')
    weekday = date.weekday()

    if weekday >= 5:  # 周末
        return False, f"周末（星期{'六' if weekday == 5 else '日'}）"

    # 检查法定节假日...
    return True, "交易日"
